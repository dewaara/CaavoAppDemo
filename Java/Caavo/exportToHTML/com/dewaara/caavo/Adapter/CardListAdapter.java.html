17   import java.util.List;

19    
20   public class CardListAdapter extends RecyclerView.Adapter<CardListAdapter.MyViewHolder> { 
21    
22      private Context context; 
23      private List<Item> list; 
24    
25      public CardListAdapter(Context context, List<Item> list) { 
26          this.context = context; 
27          this.list = list; 
28      } 
29    
30       @NonNull 
31       @Override 
32       public MyViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
33    
34           View itemView = LayoutInflater.from(parent.getContext()) 
35                  .inflate(R.layout.card_list_item,parent, false); 
36           return new MyViewHolder(itemView); 
37       } 
38    
39       @Override 
40       public void onBindViewHolder(@NonNull MyViewHolder holder, int position) { 
41    
42          Item item = list.get(position); 
43          holder.name.setText(item.getName()); 
44          holder.description.setText(item.getDescription()); 
45          holder.price.setText(item.getPrice()); 
46           Picasso.with(context) 
47                   .load(item.getThumbnail()) 
48                   .into(holder.thumbnail); 
49       } 
50    
51       @Override 
52       public int getItemCount() { 
53           return list.size(); 
54       } 
55    
56       public void removeItem(int position) 
57       { 
58           list.remove(position); 
59           notifyItemRemoved(position); 
60       } 
61    
62       public void restoreItem(Item item, int position) 
63       { 
64           list.add(position,item); 
65           notifyItemInserted(position); 
66       } 
67    
68    
69    
70       public class MyViewHolder extends RecyclerView.ViewHolder { 
71    
72           public TextView name,description,price; 
73           public ImageView thumbnail; 
74           public RelativeLayout viewBackground,viewForeground; 
75    
76    
77           public MyViewHolder(View itemView) { 
78               super(itemView); 
79               name = itemView.findViewById(R.id.name); 
80               description = itemView.findViewById(R.id.description); 
81               price = itemView.findViewById(R.id.price); 
82               thumbnail = itemView.findViewById(R.id.thumbnail); 
83               viewBackground = itemView.findViewById(R.id.view_background); 
84               viewForeground = itemView.findViewById(R.id.view_foreground); 
85           } 
86       } 
87   }
